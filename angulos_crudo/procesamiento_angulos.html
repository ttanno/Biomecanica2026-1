% FUNCIONES
function theta = calcAngles(Ox, Oy, Ax, Ay, Bx, By)
%% Vectores OA y OB
OAx = Ax - Ox;
OAy = Ay - Oy;
OBx = Bx - Ox;
OBy = By - Oy;
%% Ángulo original
dotprod = OAx.*OBx + OAy.*OBy;
normOA = sqrt(OAx.^2 + OAy.^2);
normOB = sqrt(OBx.^2 + OBy.^2);
theta = acosd(dotprod ./ (normOA .* normOB));
 
end
